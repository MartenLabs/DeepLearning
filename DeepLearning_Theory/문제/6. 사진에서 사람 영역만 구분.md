def conv2d_block(x, channel):

x = Conv2D(filters=channel, kernel_size=3, padding='same')(x)

x = BatchNormalization()(x)

x = Activation('relu')(x)

  

x = Conv2D(filters=channel, kernel_size=3, padding='same')(x)

x = BatchNormalization()(x)

x = Activation('relu')(x)

  

return x
